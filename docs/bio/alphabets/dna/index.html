Implementation of the DNA alphabet.

Example

use bio::alphabets;
let alphabet = alphabets::dna::alphabet();
assert!(alphabet.is_word(b"GATTACA"));
assert!(alphabet.is_word(b"gattaca"));
assert!(!alphabet.is_word(b"ACGU"));
